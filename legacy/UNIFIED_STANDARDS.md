### 2.1 量化标准

### 2.2 判定流程

```python
def determine_regime(mom_6m_avg, yoy, trend):
    """
    mom_6m_avg: 近6月环比均值
    yoy: 当前同比
    trend: 近6月环比趋势(斜率)
    """
    if mom_6m_avg > 100.3 and yoy > 100 and trend > 0:
        return "R1 上行"
    elif mom_6m_avg >= 99.8 and mom_6m_avg <= 100.3 and trend >= 0:
        return "R3 筑底"
    elif mom_6m_avg < 99.8 or trend < -0.1:
        return "R2 下行"
    else:
        return "R4 分化"
```

### 4.2 估算方法

```python
# 租售比 = 年租金 / 房价 × 100%
# 假设: 80㎡房产，各城市经验租金
rent_data = {
    '上海': 7500,   # 元/月
    '北京': 6800,
    '广州': 5500,
    '深圳': 5600,
    '杭州': 4700,
    '成都': 3500,
    '武汉': 3200,
    '郑州': 2800,
    '西安': 2500,
    '长春': 2200,
}

price_data = {
    '上海': 60000,  # 元/㎡
    '北京': 60000,
    '广州': 38000,
    '深圳': 55000,
    '杭州': 35000,
    '成都': 19000,
    '武汉': 18000,
    '郑州': 15000,
    '西安': 16000,
    '长春': 10000,
}

# 租售比 = (月租金×12) / (房价×面积) × 100%
for city in rent_data:
    annual_rent = rent_data[city] * 12
    total_price = price_data[city] * 80
    yield_rate = annual_rent / total_price * 100
    print(f"{city}: {yield_rate:.2f}%")
```
